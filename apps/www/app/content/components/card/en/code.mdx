## Usage

```tsx
import { Card } from '@digdir/designsystemet-react';

<Card>
  <Card.Block>Title</Card.Block>
  <Card.Block>Content</Card.Block>
  <Card.Block>Optional footer</Card.Block>
</Card>;
```
