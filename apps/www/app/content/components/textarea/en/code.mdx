## Use

```tsx
import { Textarea, Label } from '@digdir/designsystemet-react';

<>
  <Label htmlFor="my-textarea">Label</Label>
  <Textarea id="my-textarea" />
</>
```

## HTML
Apply the class name `ds-input` to `<textarea>`.

```html
<textarea class="ds-input"></textarea>
```
