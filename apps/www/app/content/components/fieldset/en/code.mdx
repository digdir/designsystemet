## Usage

```tsx
import { Fieldset, Radio } from '@digdir/designsystemet-react';

<Fieldset>
  <Fieldset.Legend>Question</Fieldset.Legend>
  <Fieldset.Description>Description</Fieldset.Description>
  <Radio label="Option 1" name="radio" value="1" />
  <Radio label="Option 2" name="radio" value="2" />
</Fieldset>
```
