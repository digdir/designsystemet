## Usage

```tsx
import { Dialog, Heading, Paragraph } from '@digdir/designsystemet-react';

<Dialog.TriggerContext>
  <Dialog.Trigger>Open Dialog</Dialog.Trigger>
  <Dialog>
    <Heading>Dialog header</Heading>
    <Paragraph>Dialog content</Paragraph>
  </Dialog>
</Dialog.TriggerContext>
```
