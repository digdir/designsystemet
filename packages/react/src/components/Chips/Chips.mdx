## Bruk

```tsx
import { Chips } from '@digdir/design-system-react';

const options = ['Norsk', 'Dansk', 'Svensk', 'Tysk', 'Spansk'];

<Chips size='xsmall'>
  {options.map((option) => (
    <Chips.Removable onClick={delete()}> {option} </Chips.Removable>
  ))}
</Chips>;

<Chips>
  {options.map((option) => (<Chips.Toggle selected={selected}>{option}</Chips.Toggle>;))}
</Chips>;
```
